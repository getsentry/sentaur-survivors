-DCMAKE_MAKE_PROGRAM=$NINJA

-DCMAKE_RUNTIME_OUTPUT_DIRECTORY=$PROJECT/unityLibrary/build/intermediates/cxx/RelWithDebInfo/$HASH/obj/$ABI
-DCMAKE_BUILD_TYPE=RelWithDebInfo
-DCMAKE_FIND_ROOT_PATH=/Users/bitfox/Workspace/demos/sentaur-survivors/.utmp/RelWithDebInfo/$HASH/prefab/$ABI/prefab
